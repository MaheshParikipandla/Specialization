public class Main
{
	 public static void main(String[] args) 
	 {
       int count=0;
        for(int i=20;(i<=50 && count<5);i++) {
            if( isPrime(i) == true) 
            {
                count++;
                System.out.println(i+ " ");
               
            }
                
        }
        

    } 
    
    static boolean isPrime(int n){

        boolean res = false;
        int i;
        for(i=2;(i<=n/2) && (n % i != 0);i++);

        if(i>n/2) {
            res = true;
        }
        return res;
    }
	
}
